# Neon Database - Example .env Configuration

# ===========================================
# STEP 1: Get your Neon connection string
# ===========================================
# 1. Go to: https://console.neon.tech
# 2. Sign up (free, no credit card needed)
# 3. Click "Create Project"
# 4. Copy the connection string from the dashboard
# 5. Paste it below

DATABASE_URL=postgresql://neondb_owner:YOUR_PASSWORD@ep-xxxxx.us-east-2.aws.neon.tech/neondb?sslmode=require

# ===========================================
# STEP 2: Set other required variables
# ===========================================

# Any random string - used to encrypt sessions
SESSION_SECRET=your_random_secret_key_change_this

# Get from: https://platform.openai.com/api-keys
AI_INTEGRATIONS_OPENAI_API_KEY=sk-your-openai-api-key-here
AI_INTEGRATIONS_OPENAI_BASE_URL=https://api.openai.com/v1

# Optional: Change model (default: gpt-4o-mini)
OPENAI_MODEL=gpt-4o-mini

# ===========================================
# STEP 3: Save and run setup
# ===========================================
# 1. Save this file as: .env
# 2. Run: npm run db:push
# 3. Run: npm run db:seed (optional, creates demo account)
# 4. Run: npm run dev
#
